    public void nhapdanhsach()

public void xeptheoID()
    { 
        System.out.println("Bạn đã chọn chức năng săp xếp theo ID");
        Comparator<hanghoa> objID = new Comparator<hanghoa>()
        {
        @Override
            public int compare(hanghoa t1, hanghoa t2) 
            {
            if(t1.ID > t2.ID)
                return 1;
            return -1;
            }
        };
        Collections.sort(list, objID);
        System.out.println(list);// sắp xếp + in 
    }



public void XepTheoTen()
    {
    System.out.println("Bạn đã chọn chức năng sắp xếp theo tên");
    Comparator<hanghoa> objname = new Comparator<hanghoa>()
        {
        @Override
        public int compare(hanghoa t, hanghoa t1) 
                {
                return t.getName(). compareTo(t1.getName());
                }
        };
        Collections.sort(list, objname);
        System.out.println(list);
    }

public hanghoa searchID(int x)
{
    for(hanghoa hh : list) //duyệt danh sách hàng hóa với mỗi phần tử là hh
    {
        if(hh.getID() == x)
            return hh;
    }
    return null;
}



public hanghoa timtensp(String name) 
    {
    for(hanghoa hh : list) 
    {
        if(hh.getName().compareToIgnoreCase(name)==0)
            return hh;
    }
    return null;
}
